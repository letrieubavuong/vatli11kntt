\begin{dang}{Lí thuyết}
\begin{itemize}
\item Phân loại sóng cơ dựa theo mối quan hệ giữa \textit{phương dao động} và \textit{phương truyền sóng}.
\item Khi sóng lan truyền đi thì \textit{năng lượng} và \textit{pha dao động} được truyền đi, còn các phần tử vật chất của môi trường chỉ dao động quanh vị trí cân bằng của chúng mà \textit{không chuyển dời} đi theo sóng.
\item Tần số của sóng chính là tần số của nguồn. Chính vì vậy khi sóng lan truyền qua các môi trường khác nhau thì tần số không bị thay đổi.
\end{itemize}
\end{dang}
