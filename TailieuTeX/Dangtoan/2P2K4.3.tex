\begin{dang}{Công suất nguồn âm – khoảng cách tới nguồn âm}
\Anlythuyet{
\begin{itemize}
\item Với sóng âm hình cầu: $\boder{S = 4\pi.r^2\Rightarrow I=\dfrac{P}{S}=\dfrac{P}{4\pi.r^2}}$
\item Xét trường hợp tổng quát với nguồn âm gồm n nguồn âm giống hệt nhau có công suất $P_0$ thì công suất bộ nguồn là: $P = nP_0$.
\begin{align*}
I=I_0.10^L=\dfrac{nP_0}{4\pi r^2}
\end{align*}
\lefthand \ Tuỳ vào từng bài toán ta xác định các thông số cố định và thay đổi rồi lập tỉ lệ để giải quyết. 
\end{itemize}
}
\end{dang}
