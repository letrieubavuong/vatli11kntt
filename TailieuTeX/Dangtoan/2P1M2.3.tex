\begin{dang}{CÔNG THỨC ĐỘC LẬP THỜI GIAN}
\Anlythuyet{\begin{itemize}
\item Công thức độc lập thời gian giữa x và v:
\begin{align*}
\boder{\left(\dfrac{x}{A}\right)^2+\left(\dfrac{v}{v_{\max}}\right)^2=1\Leftrightarrow A^2=x^2+\dfrac{v^2}{\omega^2}}
\end{align*}
\item Công thức độc lập thời gian giữa v và a: 
\begin{align*}
\boder{\left(\dfrac{v}{v_{\max}}\right)^2+\left(\dfrac{a}{a_{\max}}\right)^2=1\Leftrightarrow A^2=\dfrac{v^2}{\omega^2}+\dfrac{a^2}{\omega^4}}
\end{align*}
\item Công thức độc lập thời gian giữa a và x: \boder{a=-\omega^2x}
\end{itemize}}
\end{dang}
