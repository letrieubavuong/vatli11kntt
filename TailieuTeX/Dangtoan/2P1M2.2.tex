\begin{dang}{VẬN TỐC VÀ GIA TỐC}
%\Anlythuyet{\begin{itemize}
%\item Để tìm vận tốc tại thời điểm t ta làm như sau:
%\begin{itemize}
%\item Từ phương trình li độ suy ra phương trình vận tốc.
%\item Thay t vào phương trình vận tốc.
%\end{itemize}
%\item Để tìm gia tốc tại thời điểm t ta làm như sau:
%\begin{itemize}
%\item Thay t vào phương trình li độ để tìm li độ.
%\item Dùng công thức $a=-\omega^2x$ để tìm gia tốc.
%\end{itemize}
%\end{itemize}}
\end{dang}
