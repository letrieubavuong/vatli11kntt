\begin{dang}{Bài toán liên quan đến phương trình sóng}
\Anlythuyet{
\begin{itemize}
\item Tính tốc độ truyền sóng bằng cách đồng nhất phương trình hoặc dùng công thức tính nhanh: 
\begin{align*}
\boder{v = \dfrac{\text{Hệ số của t}}{\text{Hệ số của x}}}
\end{align*}
\item Để tìm li độ, biên độ hoặc vận tốc sóng tại một thời điểm ta thay các đại lượng đề cho vào phương trình li độ hoặc phương trình vận tốc. 
\end{itemize}
}
\end{dang}
