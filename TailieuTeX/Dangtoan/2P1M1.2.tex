	\begin{dang}{CÁC ĐẠI LƯỢNG CƠ BẢN TRONG DAO ĐỘNG ĐIỀU HÒA}
\Anlythuyet{\begin{itemize}
\item Chiều dài quỹ đạo: $L=2A$.
\item Quãng đường đi trong một chu kì: $S_T=4A$.
\item Để xác định chính xác các đại lượng $A,\ \omega,\ \varphi$ ta tiến hành chuẩn hoá phương trình dao động điều hoà về dạng chuẩn: 
\begin{align*}
x=A\cos \left(\omega t+\varphi\right)\ \text{với}\ 
\begin{cases}
A,\ \omega >0\\ 
-\pi <\varphi \leq \pi 
\end{cases}
\end{align*}
\end{itemize}
\Bookmark \textbf{Một số công thức lượng giác cần nhớ}
\begin{enumEX}[\faCheckCircleO]{2}
\item $\sin \alpha = \cos \left(\alpha-\dfrac{\pi}{2}\right)$.
\item $-\sin \alpha=\cos \left(\alpha + \dfrac{\pi}{2}\right)$.
\item $-\cos \alpha =\cos \left(\alpha \pm \pi\right)$.
\item $\cos \alpha=\cos \left(-\alpha\right)$.
\item $\cos \alpha =\cos (\alpha \pm k2\pi)$
\item $\sin \alpha =\sin (\alpha \pm k2\pi)$ 
\end{enumEX}}
\end{dang}
