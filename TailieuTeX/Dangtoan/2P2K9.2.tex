\begin{dang}{Xác định vị trí vân sáng – vân tối – khoảng vân}
\Anlythuyet{
\begin{itemize}
\item Khoảng cách giữa hai vân M, N bất kì được xác định theo biểu thức:
\begin{align*}
\boder{\Delta x=\abs{x_M-x_N}}\quad 
\begin{cases}
\ct{M và N cùng phía với vân trung tâm thì}\ x_M\ \ct{và}\ x_N\  \ct{cùng dấu}.\\
\ct{M và N khác phía với vân trung tâm thì}\ x_M\ \ct{và}\  x_N\  \ct{khác dấu}.
\end{cases}
\end{align*}
\item Khoảng cách giữa n vân sáng (tối) liên tiếp là: \boder{(n-1).i}
\item Xác định điểm M thuộc vân sáng hay vân tối:\\
\quad$\begin{cases}
\ct{M thuộc vân sáng nếu}\ x_M=k.i.\\
\ct{M thuộc vân tối nếu}\ x_M=(k+0,5).i.
\end{cases}$
\end{itemize}
}
\end{dang}
