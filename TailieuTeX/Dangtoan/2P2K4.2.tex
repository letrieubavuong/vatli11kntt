\begin{dang}{Sự truyền âm – các đặc trưng của âm}
\Anlythuyet{
\begin{itemize}
\item Một người nghe hai âm truyền từ một nguồn trong hai môi trường cách nhau một khoảng thời gian $\Delta t$.
\begin{itemize}
\item Thời gian âm truyền trong môi trường thứ nhất:$t_1=\dfrac{\ell}{v_1}$.
\item Thời gian âm truyền trong môi trường thứ hai:$t_2=\dfrac{\ell}{v_2}$.
\item Người này nghe hai âm cách nhau: $\Delta t=|t_2-t_1|=\left|\dfrac{\ell}{v_2}-\dfrac{\ell}{v_1}\right|$. 
\end{itemize}
\item Khi thả một vật rơi thì quãng đường rơi: $s=\dfrac{1}{2}gt^2$.
\end{itemize} 
}
\end{dang}
