\begin{dang}{Lí thuyết}
\Anlythuyet{
\begin{itemize}
\item Nhiệt độ môi trường càng cao và tính đàn hồi càng lớn thì tốc độ truyền âm càng lớn.
\item Khi âm truyền từ môi trường này sang môi trường khác thì vận tốc truyền âm thay đổi, bước sóng của sóng âm thay đổi còn tần số của âm thì \textit{không thay đổi}.
\end{itemize}
}
\end{dang}
