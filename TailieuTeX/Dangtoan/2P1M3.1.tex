\begin{dang}{VIẾT PHƯƠNG TRÌNH DAO ĐỘNG}
\begin{enumerate}
\item Tìm tần số góc \bth{\omega}: \bth{\omega=2\pi f=\dfrac{2\pi}{T}=\dfrac{v_{max}}{A}=\sqrt{\dfrac{a_{max}}{A}}=\sqrt{\dfrac{k}{m}}}
\item Tìm biên độ dao động A: 
\begin{itemize}
\item $A=\dfrac{L}{2}=\dfrac{S_T}{4}=\dfrac{v_{max}}{\omega}=\dfrac{a_{max}}{\omega^2}$
\quad || \bth{A=\sqrt{x^2+\dfrac{v^2}{\omega^2}}=\sqrt{\dfrac{v^2}{\omega^2}+\dfrac{a^2}{\omega^4}}}
\end{itemize}
\item Tìm $\varphi$: dựa vào điều kiện ban đầu 
\begin{itemize}
\item Giải phương trình lượng giác: \bth{t=0\Rightarrow\vlbrace{\begin{array}{ll}
x_0 &=A\cos\varphi \\ 
v_0 &=-A\omega\sin\varphi
\end{array} } \Rightarrow \varphi=?}
\end{itemize}
\item Thay vào phương trình dao động tổng quát: \bth{x=A\cos(\omega t+\varphi)}
\end{enumerate}
\end{dang}
