\paragraph{Độ lệch pha giữa hai điểm M và N}
\begin{itemize}
\item Độ lệch pha giữa hai điểm M và N bất kì:  \boder{\Delta\varphi=2\pi\dfrac{x_N-x_M}{\lambda}}
\item Độ lệch pha giữa hai điểm M và N trên một phương truyền sóng: \boder{\Delta\varphi=2\pi\dfrac{d}{\lambda}}
\end{itemize}
