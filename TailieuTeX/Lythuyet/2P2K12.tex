\subsubsection{Sự phản xạ sóng}
\begin{itemize}
\item Với đầu phản xạ cố định: sóng tới và sóng phản xạ \textit{ngược pha} tại điểm phản xạ.
\item Với đầu phản xạ tự do: sóng tới và sóng phản xạ \textit{cùng pha} tại điểm phản xạ.
\end{itemize}
\subsubsection{Định nghĩa}
\newghichu{
\begin{itemize}
\item Sóng có các nút sóng và bụng sóng cố định trong không gian gọi là \textbf{sóng dừng}.
\item Những điểm luôn luôn đứng yên gọi là \textbf{\textit{nút}}.
\item Những điểm luôn luôn dao động với biên độ lớn nhất là \textbf{\textit{bụng}}.
\end{itemize}}
\Note{Sóng dừng thường là giao thoa giữa sóng tới và sóng phản xạ}

\subsubsection{Điều kiện sóng dừng}
\begin{itemize}
\item Dây cố định hai đầu: \bth{\boder{\ell=k.\dfrac{\lambda}{2}}} với $k = 1,\ 2,\ 3...\Rightarrow\begin{cases}
\text{số bụng}=k\\
\text{số nút}=k+1\\
\end{cases}$
\newghichu{\textit{Để có sóng dừng trên dây hai đầu cố định thì chiều dài dây phải bằng số nguyên lần nửa bước sóng}}
\item Một đầu dây cố định, một đầu tự do: \bth{\boder{\ell=(2k+1).\dfrac{\lambda}{4}}} với $k = 0,\ 1,\ 2,\ 3....\Rightarrow\begin{cases}
\text{số bụng}=k+1\\
\text{số nút}=k+1\\
\end{cases}$
\newghichu{\textit{Để có sóng dừng trên dây một đầu cố định, một đầu tự do thì chiều dài dây phải bằng số lẻ lần một phần tư bước sóng}}
\end{itemize}
\subsubsection{Một số lưu ý}
\begin{itemize}
\item Sóng tới và sóng phản xạ luôn cùng tần số.
\item Thời gian giữa hai lần sợi dây duỗi thẳng là $\dfrac{T}{2}$.
\item Hai điểm đối xứng nhau qua nút luôn dao động ngược pha, hai điểm đối xứng nhau qua bụng dao động cùng pha.
\item Các điểm nằm trên cùng một bó sóng thì dao động cùng pha.
\item Các điểm nằm trên hai bó sóng liền kề thì dao động ngược pha nhau.
\item Các điểm nằm trên bó cùng chẵn hoặc cùng lẻ dao động cùng pha, các điểm nằm trên bó lẻ thì dao động ngược pha với các điểm nằm trên bó chẵn.
\end{itemize}
