\subsubsection{Vận tốc của vật dao động điều hòa}
\centerline{\bth{\boder{v=-\omega A\sin(\omega t+\varphi)=\omega A\cos\pbrace{\omega t+\varphi+\dfrac{\pi}{2}}
}}}
\begin{itemize}
\item Vận tốc cực đại: \ \ $v_{\max}=A\omega$\ \ \ khi vật qua vị trí $x=0$ (vị trí cân bằng) theo chiều dương.
\item Vận tốc cực tiểu:\ \  $v_{\min}=-A\omega$\ khi vật qua vị trí $x=0$ (vị trí cân bằng) theo chiều âm.
\item Tốc độ cực đại: \ \ \ $\left|v_{\max}\right|=\omega A$\ \ khi vật qua vị trí $x=0$ (vị trí cân bằng).
\item Tốc độ cực tiểu: \ \ $\left|v_{\min}\right|=0\ $ $\ \ $ khi vật qua vị trí $x=\pm A$ (vị trí biên).
\end{itemize}
\begin{note}
\begin{itemize}
\item Vectơ vận tốc $\vec{v}$ cùng hướng với chuyển động. 
\item Từ vị trí cân bằng ra biên vật chuyển động chậm dần (tốc độ giảm). 
\item Từ biên về vị trí cân bằng vật chuyển động nhanh dần (tốc độ tăng).
\end{itemize}
\end{note}
\subsubsection{Gia tốc của vật dao động điều hòa}
\centerline{\bth{\boder{a=-\omega^2A\cos(\omega t+\varphi)=\omega^2A\cos(\omega t+\varphi+\pi)=-\omega^2x
}}}
\begin{itemize}
\item Gia tốc cực đại:$\  \ \ \ \ \ \ \ \ \ \ \ $ $a_{\max}=A\omega^2$ $\ \ $\ \  tại vị trí $x=-A$ (vị trí biên âm).
\item Gia tốc cực tiểu:$\  \ \ \ \ \ \ \ \ \ \ $ $a_{\min}=-A\omega^2$ $\ $ tại vị trí $x=A$ (vị trí biên dương).
\item Độ lớn gia tốc cực đại:$\ $ $\left|a_{\max}\right|=\omega^2A$ tại vị trí $x=\pm A$ (vị trí biên).
\item Độ lớn gia tốc cực tiểu: $\left|a_{\min}\right|=0$ $\ \ \ \ \ \ $ tại vị trí $x=0$ (vị trí cân bằng).
\end{itemize}
\begin{note}
\begin{itemize}
\item Vectơ gia tốc $\vec{a}$ luôn hướng về vị trí cân bằng.
\item Độ lớn gia tốc tỉ lệ với độ lớn li độ của vật.
\end{itemize}
\end{note}
